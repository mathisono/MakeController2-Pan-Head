00061    per active UDP "connection". */
00062 #define MEMP_NUM_UDP_PCB        NETWORK_UDP_CONNS //8

00064    connections. */
00065 #define MEMP_NUM_TCP_PCB        NETWORK_TCP_CONNS //10

00067    connections. */
00068 #define MEMP_NUM_TCP_PCB_LISTEN NETWORK_TCP_LISTEN_CONNS //8
00069 /* MEMP_NUM_TCP_SEG: the number of simultaneously queued TCP
